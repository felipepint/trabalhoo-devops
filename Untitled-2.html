Tutorial Interativo de JavaScript

1. Declaração de Variáveis

let nome = "João";
const idade = 25;
var ativo = true;

2. Tipos de Dados

  • String: "texto"
  • Number: 10, 3.14
  • Boolean: true, false
  • Array: [1, 2, 3]
  • Object: { nome: "Ana", idade: 30 }

3. Operadores

// Aritméticos
let soma = 5 + 3;
let produto = 4 * 2;

// Comparação
5 > 3      // true
5 === "5"  // false

// Lógicos
true && false  // false
true || false  // true

4. Condicionais

const idade = 18;
if (idade >= 18) {
  console.log("Você é maior de idade.");
} else {
  console.log("Você é menor de idade.");
}

5. Loops

// Loop for
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// Loop while
let x = 0;
while (x < 3) {
  console.log("x é " + x);
  x++;
}

6. Arrays e Métodos

const frutas = ["maçã", "banana", "laranja"];
frutas.push("uva");
frutas.pop();
console.log(frutas.length);

7. Objetos

const pessoa = {
  nome: "Carlos",
  idade: 30,
  saudacao: function() {
    return "Olá, " + this.nome;
  }
};

console.log(pessoa.nome);
console.log(pessoa.saudacao());

Função Declarada

function saudacao(nome) {
  return "Olá, " + nome + "!";
}

Arrow Function

const dobro = x => x * 2;

Função como parâmetro

function calcular(a, b, operacao) {
  return operacao(a, b);
}

console.log(calcular(4, 2, (x, y) => x * y));

Manipulando o DOM

const el = document.getElementById("demo");
el.innerText = "Novo texto!";
